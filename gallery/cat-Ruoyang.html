def heart(size,color): #reusable heart
  pencolor(color)
  fillcolor(color)
  begin_fill()
  left(150)
  forward(9*size)
  circle(-9*size,200)
  left(120)
  circle(-9*size,200)
  forward(18*size)
  end_fill()

penup() #heart
goto(20,20)
pendown()
pencolor("red")
fillcolor("red")
heart(1,"red")

def star(size, color): #reusable star
  pencolor(color)
  fillcolor(color)
  begin_fill()
  for _ in range(5):
    forward(size)
    right(144)
    forward(size)
    left(72)
  end_fill()


for i in range(12): #circle of stars
  penup()
  goto(20,-30)
  setheading(i*30+90)
  forward(120)
  pendown()
  left(54)
  star(15,"yellow")
